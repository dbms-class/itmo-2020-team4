Table Facilities as F{
  id int [pk, increment]
  address varchar
  n_objects int
  function varchar
  name varchar  [default: null]
}

Table Sportsmen as S {
  id int [pk, increment] // auto-increment
  full_name varchar
  delegation_id int 
  place int [ref: > F.id]
  
  // and many more
}

Table Delegations as D{
  id int [pk, increment]
  country varchar
  head_name varchar [unique]
  head_phone varchar [unique]
  place int [ref: > F.id]
 }
